%* Paramètres de matrices du TP de MNB foireux mais qui marchent, permet de redefinir les espacements entre coeffs, utile pour matrices enormes

\makeatletter
\renewcommand*\env@matrix[1][\arraystretch]{%
  \edef\arraystretch{#1}%
  \hskip -\arraycolsep
  \let\@ifnextchar\new@ifnextchar
  \array{*\c@MaxMatrixCols c}}
\makeatother
